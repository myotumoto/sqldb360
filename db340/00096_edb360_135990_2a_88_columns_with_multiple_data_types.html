   AND t.table_name NOT LIKE 'BIN%'
), columns AS (

       --c.con_id,
       c.owner,
       c.table_name,
       c.column_name,
       c.data_type
  FROM dba_tab_columns c
 WHERE c.owner NOT IN ('ANONYMOUS','APEX_030200','APEX_040000','APEX_040200','APEX_180200','APEX_SSO','APPQOSSYS','CTXSYS','DBSNMP','DIP','EXFSYS','FLOWS_FILES','MDSYS','OLAPSYS','ORACLE_OCM','ORDDATA','ORDPLUGINS','ORDSYS','OUTLN','OWBSYS')

   AND c.data_type != 'UNDEFINED'

   AND c.data_type != 'UNDEFINED'
), table_columns AS (

       --c.con_id,

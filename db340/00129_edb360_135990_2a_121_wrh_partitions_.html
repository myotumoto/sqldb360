-- from http://jhdba.wordpress.com/tag/purge-wrh-tables/
WITH x AS (
SELECT --con_id,
       table_name, count(*)
  FROM dba_tab_partitions
 WHERE table_name like 'WRH$%'
   AND table_owner = 'SYS'
group by
       --con_id,
       table_name
)
SELECT x.*
       --,c.name con_name
  FROM x
       --LEFT OUTER JOIN v$containers c ON c.con_id = x.con_id
 ORDER BY
       --x.con_id,
       x.table_name;
